class pygame.Rect {}
class numpy.matrix {}

enum Tile {
    + empty
    + ground
    + wall
    + warp
}

class Editor {
    + on_click(x: int, y: int): None
    + draw(): None
}

class Grid {
    - elements: numpy.matrix<64, 64, Tile>

    + draw(): None
}

class Palette {
    - {static} elements: dict[pygame.Rect, Tile]
    - {static} selected: pygame.Rect

    + {static} in_palette(x: int, y: int): bool
    + {static} select(x: int, y: int): None
    + {static} selected(): Tile
    + {static} draw(): None
}

Grid --> Tile
Grid --> numpy.matrix

Palette --> Tile
Palette --> pygame.Rect

Editor -- Palette
Editor -- Grid
